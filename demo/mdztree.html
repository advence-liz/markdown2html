const gulp = require('gulp');
const markdown = require('gulp-markdown');
const marked = require('marked');

/**
 * default task
 */

gulp.task('default', function () {
    return gulp.src('src/*.md')
        .pipe(markdown({
            highlight: function (code, lang) {
                pygmentize({ lang: lang, format: 'html' }, code)
            }
        }))

        .pipe(gulp.dest('dist'));
});
